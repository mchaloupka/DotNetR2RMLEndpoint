prefix sukl: <http://opendata.sukl.cz/dlp/>

SELECT *
WHERE {
   {
      ?id a sukl:medicinalProduct;
         sukl:hasAddiction ?a;
         sukl:label ?label;
         sukl:package ?package;
         sukl:hasWay ?w.
      ?w sukl:label ?way.
      ?a sukl:label ?add;
         sukl:id "E".
   }
   UNION
   {
      ?mid a sukl:medicinalProduct;
         sukl:hasAddiction ?a;
         sukl:hasComposition ?composition.
      ?a sukl:id "E".
      ?composition sukl:contains ?i.
      ?i sukl:label ?ingredient.
   }
}
